    –°–ª–æ–∂–∏–º —É—Ä–∞–≤–Ω–µ–Ω–∏—è (1) –∏ (2). –ü–æ–ª—É—á–∏–º: \[ (\ddot{œÜ_1} + \ddot{œÜ_2}) + (p_1 - p_2)(œÜ_1 + œÜ_2) = 0 \iff (\ddot{œÜ_1} + \ddot{œÜ_2}) + \frac{g}{L} (œÜ_1 + œÜ_2) = 0 \] –û–±–æ–∑–Ω–∞—á–∏–º \( œÜ_1 + œÜ_2 = œà, \space \frac{g}{L} = w_1^2 \).

    –¢–æ–≥–¥–∞, –∫–∞–∫ –±—ã–ª–æ –æ—Ç–º–µ—á–µ–Ω–æ —Ä–∞–Ω–µ–µ, —Ä–µ—à–µ–Ω–∏–µ —É—Ä–∞–≤–Ω–µ–Ω–∏—è (4) –±—É–¥–µ—Ç –∞–Ω–∞–ª–æ–≥–∏—á–Ω—ã–º: \( \eta = 2œÜ_{02} \cos{œâ_2 t} \).
    –£—á–∏—Ç—ã–≤–∞—è —Ä–∞–Ω–µ–µ –≤–≤–µ–¥—ë–Ω–Ω—É—é –∑–∞–º–µ–Ω—É, –ø–æ–ª—É—á–∏–º: \[ \begin{equation} \left\{ \begin{aligned} œÜ_1 + œÜ_2 = 2œÜ_{01} \cos{œâ_1 t}\\ œÜ_1 - œÜ_2 = 2œÜ_{02} \cos{œâ_2 t} \end{aligned} \right. \iff \left\{ \begin{aligned} œÜ_1 = œÜ_{01}\cos{œâ_1 t} + œÜ_{02}\cos{œâ_2 t}\\ œÜ_2 = œÜ_{01}\cos{œâ_1 t} - œÜ_{02}\cos{œâ_2 t} \end{aligned} \right. \end{equation}\]
    –ü—Ä–æ–¥–∏—Ñ—Ñ–µ—Ä–µ–Ω—Ü–∏—Ä–æ–≤–∞–≤ –ø–æ–ª—É—á–µ–Ω–Ω—ã–µ —Ñ—É–Ω–∫—Ü–∏–∏ –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏—è –æ—Ç –≤—Ä–µ–º–µ–Ω–∏, –ø–æ–ª—É—á–∏–º —Ñ—É–Ω–∫—Ü–∏–∏ —Å–∫–æ—Ä–æ—Å—Ç–µ–π: \[v_1 = -œÜ_{01}œâ_1\sin{œâ_1 t} - œÜ_{02}œâ_2\sin{œâ_2 t}\] \[v_1 = -œÜ_{01}œâ_1\sin{œâ_1 t} + œÜ_{02}œâ_2\sin{œâ_2 t}\]

    // –ü–∞—Ä–∞–º–µ—Ç—Ä—ã —Å–∏—Å—Ç–µ–º—ã

const L_1 = 0.5; //  —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ  L1  –æ—Ç —Ç–æ—á–∫–∏ –∫—Ä–µ–ø–ª–µ–Ω–∏—è –º–∞—è—Ç–Ω–∏–∫–æ–≤ (–º)

const m = 1; // (–º)
const beta = 0.1; // –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –∑–∞—Ç—É—Ö–∞–Ω–∏—è
const phi_01 = 0.3; // –ù–∞—á–∞–ª—å–Ω—ã–π —É–≥–æ–ª –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏—è œÜ1 (—Ä–∞–¥)
const phi_02 = 0.3; // –ù–∞—á–∞–ª—å–Ω—ã–π —É–≥–æ–ª –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏—è œÜ2 (—Ä–∞–¥)

// –ù–∞—à–∏ –æ–±–æ–∑–Ω–∞—á–µ–Ω–∏—è

    return phi_01 * Math.cos(omega_1 * t) + phi_02 * Math.cos(omega_2 * t);
}

function phi_2(t) { // –£–≥–æ–ª –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏—è –≤—Ç–æ—Ä–æ–≥–æ –º–∞—è—Ç–Ω–∏–∫–∞
    return phi_01 * Math.cos(omega_1 * t) - phi_02 * Math.cos(omega_2 * t);

}
